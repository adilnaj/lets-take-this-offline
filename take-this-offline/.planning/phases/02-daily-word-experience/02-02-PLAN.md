Both pages are async Server Components (no 'use client'). Both use `export const dynamic = 'force-dynamic'` to prevent prerender errors (consistent with Phase 1 decision for auth-dependent pages — and today's word changes daily, so it must not be statically cached).

**src/app/page.tsx**:
```tsx
import { getTodaysWord } from '@/lib/words'
import { WordCard } from '@/components/word-card'
import { SiteHeader } from '@/components/site-header'

export const dynamic = 'force-dynamic'

export async function generateMetadata() {
  const word = await getTodaysWord()
  if (!word) return { title: "Let's Take This Offline" }
  return {
    title: `${word.title} | Let's Take This Offline`,
    description: word.exec_summary,
    openGraph: {
      title: word.title,
      description: word.exec_summary,
      images: [`/api/og/${word.slug}`],
    },
    twitter: { card: 'summary_large_image' },
  }
}

export default async function HomePage() {
  const word = await getTodaysWord()
  if (!word) {
    return (
      <main>
        <SiteHeader />
        <div className="container max-w-2xl mx-auto py-12 px-4">
          <p className="text-muted-foreground">No word scheduled for today. Check back tomorrow.</p>
        </div>
      </main>
    )
  }
  return (
    <main>
      <SiteHeader />
      <div className="container max-w-2xl mx-auto py-12 px-4">
        <WordCard word={word} />
      </div>
    </main>
  )
}
```

**src/app/word/[slug]/page.tsx**:
```tsx
import { getWordBySlug } from '@/lib/words'
import { WordCard } from '@/components/word-card'
import { SiteHeader } from '@/components/site-header'
import { notFound } from 'next/navigation'

export const dynamic = 'force-dynamic'

export async function generateMetadata({ params }: { params: { slug: string } }) {
  const word = await getWordBySlug(params.slug)
  if (!word) return { title: "Not Found | Let's Take This Offline" }
  return {
    title: `${word.title} | Let's Take This Offline`,
    description: word.exec_summary,
    openGraph: {
      title: word.title,
      description: word.exec_summary,
      images: [`/api/og/${word.slug}`],
    },
    twitter: { card: 'summary_large_image' },
  }
}

export default async function WordPage({ params }: { params: { slug: string } }) {
  const word = await getWordBySlug(params.slug)
  if (!word) notFound()
  return (
    <main>
      <SiteHeader />
      <div className="container max-w-2xl mx-auto py-12 px-4">
        <WordCard word={word} />
      </div>
    </main>
  )
}
```
