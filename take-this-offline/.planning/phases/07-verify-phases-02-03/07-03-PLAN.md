```typescript
import { NextRequest, NextResponse } from 'next/server'
import Anthropic from '@anthropic-ai/sdk'
import { createClient } from '@/lib/supabase/server'

const FEEDBACK_MODEL = 'claude-haiku-4-5-20251001'
const anthropic = new Anthropic()

export async function POST(req: NextRequest) {
  const supabase = await createClient()
  const { data: { user } } = await supabase.auth.getUser()
  if (!user) return NextResponse.json({ error: 'Unauthorized' }, { status: 401 })

  const body = await req.json()
  const { wordTitle, definition, sentence } = body
  if (!wordTitle || !definition || !sentence) {
    return NextResponse.json({ error: 'Missing required fields' }, { status: 400 })
  }

  try {
    const message = await anthropic.messages.create({ ... })
    const feedback = (message.content[0] as { text: string }).text
    return NextResponse.json({ feedback })
  } catch (err) {
    console.error('[sentence-feedback] Anthropic error:', err)
    return NextResponse.json({ error: 'AI feedback unavailable' }, { status: 502 })
  }
}
```

    1. Query `sentence_feedback_calls` for the current user and today's date:
       ```typescript
       const today = new Date().toISOString().slice(0, 10) // 'YYYY-MM-DD'
       const { data: callRow } = await supabase
         .from('sentence_feedback_calls')
         .select('call_count')
         .eq('user_id', user.id)
         .eq('call_date', today)
         .single()
       ```
    2. If `callRow?.call_count >= DAILY_LIMIT`, return:
       ```typescript
       return NextResponse.json(
         { error: 'Daily feedback limit reached. Come back tomorrow!' },
         { status: 429 }
       )
       ```

    After the Anthropic call succeeds (inside the try block, after extracting `feedback` and before the return), upsert to increment the counter:
    ```typescript
    await supabase.from('sentence_feedback_calls').upsert(
      { user_id: user.id, call_date: today, call_count: 1 },
      { onConflict: 'user_id,call_date', ignoreDuplicates: false }
    )
    // Note: Supabase upsert does not support expressions like call_count + 1 via the JS client.
    // Alternative: use a raw RPC or update after the fact.
    ```

    Since the Supabase JS client upsert cannot do `call_count + EXCLUDED.call_count` directly, use a two-step approach instead: first attempt an insert (ignore conflict), then always increment:
    ```typescript
    // Ensure row exists for today
    await supabase.from('sentence_feedback_calls').upsert(
      { user_id: user.id, call_date: today, call_count: 0 },
      { onConflict: 'user_id,call_date', ignoreDuplicates: true }
    )
    // Increment
    await supabase.rpc('increment_sentence_feedback', {
      p_user_id: user.id, p_date: today
    })
    ```

    ```typescript
    await supabase.from('sentence_feedback_calls').upsert({
      user_id: user.id,
      call_date: today,
      call_count: (callRow?.call_count ?? 0) + 1,
    })
    ```
    This is safe because the route handler is invoked serially per user request. The read-then-write window is acceptable for this low-frequency daily cap (not a financial transaction).
